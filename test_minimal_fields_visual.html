    'customer_name',

foreach ($required as $field) {

        sendResponse(false, null,

async function validateForm() {
    // Customer
    if (!$('#customer-select').val()) {
        return false;
    }
    
    // Appliance
    if (!$('#appliance-select').val()) {
        return false;
    }
    
    // Date In
    if (!$('#date-in').val()) {
        return false;
    }
    
    // Status
    if (!$('select[name="status"]').val()) {
        return false;
    }
    
    return true; // Only 4 checks!
}
